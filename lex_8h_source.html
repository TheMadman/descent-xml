92inline _descent_xml_read_t _descent_xml_read(
93 struct libadt_const_lptr script,
94 descent_xml_classifier_fn *const previous
95)
96{
97 wchar_t c = 0;
98 mbstate_t mbs = { 0 };
99 _descent_xml_read_t result = { 0 };
100 result.amount = _descent_xml_mbrtowc(&c, script, &mbs);
101 if (_descent_xml_read_error(result))
102 result.type = (descent_xml_classifier_fn*)descent_xml_classifier_unexpected;
103 else
104 result.type = (descent_xml_classifier_fn*)previous(c);
105
106 result.script = libadt_const_lptr_index(script, (ssize_t)result.amount);
107 return result;
108}

136DESCENT_XML_EXPORT inline struct descent_xml_lex descent_xml_lex_next_raw(
137 struct descent_xml_lex previous
138)
139{
140 const ssize_t value_offset = (char *)previous.value.buffer
141 - (char *)previous.script.buffer;
142 struct libadt_const_lptr next = libadt_const_lptr_index(
143 previous.script,
144 value_offset + previous.value.length
145 );
146
147 _descent_xml_read_t
148 read = _descent_xml_read(next, previous.type),
149 previous_read = read;
150
151 if (_descent_xml_read_error(read))
152 return (struct descent_xml_lex) {
153 .script = previous.script,
154 .type = descent_xml_classifier_unexpected,
155 .value = libadt_const_lptr_truncate(next, 0),
156 };
157
158 if (read.type == descent_xml_classifier_eof) {
159 return (struct descent_xml_lex) {
160 .script = previous.script,
161 .type = read.type,
162 .value = libadt_const_lptr_truncate(next, read.amount)
163 };
164 }
165
166 size_t value_length = read.amount;
167 for (
168 read = _descent_xml_read(read.script, read.type);
169 !_descent_xml_read_error(read);
170 read = _descent_xml_read(read.script, read.type)
171 ) {
172 if (read.type != previous_read.type)
173 break;
174
175 previous_read = read;
176 value_length += read.amount;
177 }
178
179 return (struct descent_xml_lex) {
180 .script = previous.script,
181 .type = previous_read.type,
182 .value = libadt_const_lptr_truncate(next, value_length),
183 };
184}
